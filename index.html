🕸️ Real Web Scraper

Browsers block Google. Use this with a tiny Node.js backend.

Backend code below ↓ Copy to `server.js`

npm i express cheerio cors node-fetch
// server.js
import express from 'express'; import cheerio from 'cheerio'; 
import fetch from 'node-fetch'; import cors from 'cors';
const app = express(); app.use(cors());
app.get('/scrape', async (req,res)=>{
  const q = req.query.q;
  const html = await fetch(`https://html.duckgo.com/html/?q=${q}`).then(r=>r.text());
  const $ = cheerio.load(html);
  const results = [];
  $('.result__a').each((i,el)=>{
    results.push({title: $(el).text(), url: $(el).attr('href')});
  });
  res.json(results.slice(0,10));
});
app.listen(3000,()=>console.log('http://localhost:3000'))
